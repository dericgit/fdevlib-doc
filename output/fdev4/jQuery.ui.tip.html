ui-tip: tip提示框

介绍

tip：简单提示帮助
符合fui-tip的DOM结构和CSS。

开始


$.use('fui-tip',function(){
	$("...").tip({
		graveId:""                      /* 预埋容器的ID */
		isRoundedCorner:true,           /* 是否圆角 --*/
		closeButton: false,             /* 是否有关闭按钮 */
		onCloseButtonClick:$.noop,      /* 当有关闭按钮时,点击关闭按钮时回调的业务的方法*/
		txt: 'Message',                 /* 提示文案 -- */
		local: 1,                       /* tips 浮动的相对位置 1:top left 2:top right 3:bottom left 4:bottom right 5:left top 6:left bottom 7:right top 8:right bottom -- */
		arrow: 9,                       /* tips的arrow相对位置 0:null 1:top left 2:top right 3:bottom left 4:bottom right 5:left top 6:left bottom 7:right top 8:right bottom 9:default -- */
		width: 230,                     /* tips 宽度 -- */
		dLeft: 0,                       /* 向左微调坐标 -- */
		dTop: 0,                        /* 向上微调坐标 -- */
		isAnim: true,                   /* 动画弹动效果是否开启 -- */
		isHold: true,                   /* 鼠标经过时是否hold -- */
		tipsHold: true,                 /* 鼠标经过tips固定tips */
		keep: 200,                      /* 显示持续时间，单位毫秒 -- */
		isOnloadShow: false,            /* 页面onload时是否显示 -- */
		onloadHold: 5000,               /* onload时显示持续时间 -- */
		overflowChange: true,           /* 超出边框是否反向处理 -- */
		eListener: 'mouseenter',
		classPrefix : 'fui'             /* 若使用fui-tip,此选项必须配为 'fui' */
	});
});

Configs - 基础配置

graveId : string
Default : "aissaSimpleTipsBox"
动态生成预埋容器id
isRoundedCorner : boolean
Default : true
是否有圆角
closeButton : boolean
Default : false
是否有关闭按钮
oncloseButtonClick : function
Default : 空方法
当点击关闭按钮时回调的方法
txt : string
Default : "Message"
提示时显示文案，也可是html结构
local : int{1 ~ 8}
Default : 1
tips 浮动的相对位置 1:top left 2:top right 3:bottom left 4:bottom right 5:left top 6:left bottom 7:right top 8:right bottom
arrow : int{0 ~ 9}
Default : 9
tips的arrow相对位置 0:null 1:top left 2:top right 3:bottom left 4:bottom right 5:left top 6:left bottom 7:right top 8:right bottom 9:default
width : int
Default : 64
tips 宽度数值
dLeft : int
Default : 0
向左微调坐标像素数值
dTop : int
Default : 0
向上微调坐标像素数值
isAnim : boolean
Default : true
动画弹动效果是否开启
isHold : boolean
Default : true
鼠标经过时是否hold
tipsHold : boolean
Default : true
鼠标移出tips后是否继续hold tips
keep : int
Default : 200
tipsHold后持续显示时间，单位毫秒
isOnloadShow : boolean
Default : false
页面onload成功后是否马上显示
onloadHold : int
Default : 5000
isOnloadShow后持续显示时间，单位毫秒
overflowChange : boolean
Default : true
超出边框是否反向处理
showListener : string
Default : "mouseenter"
显示tip的触发方式 "mouseover" | "click" | ... 当配置为空字符串或null时，由用户调用接口来显示tip
hideListener : string
Default : "mouseleave"
隐藏tip的触发方式 "mouseleave" | "click" | ... 当配置为空字符串或null时，由用户调用接口来隐藏tip
