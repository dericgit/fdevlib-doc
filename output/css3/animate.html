Animate.css

介绍

Animate.css包含了一堆很酷、有趣、跨浏览器的动画，你可以方便在项目中使用。其中包含弹入弹出、渐现渐隐、滑入滑出、翻转、旋转、光速效果等动画

最好配合modernizr.js一起使用，实现渐进增强效果。

开始

引入CSS

// min文件
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/animate-min.css");

源文件

@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/core.css"); // 核心           
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/attention.css"); // Attention Seekers 
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/bouncein.css"); // BounceIn(弹入)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/bounceout.css"); // BounceOut(弹出)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/fadein.css"); // FadeIn(渐现)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/fadeout.css"); // FadeOut(渐隐) 
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/slider.css"); // Slider(滑入滑出)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/flip.css"); // Flip(翻转)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/rotatein.css"); // RotateIn(旋转进入) 
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/rotateout.css"); // RotateOut(旋转消失)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/lightspeed.css"); // LightSpeed(光速效果)
@import url("http://style.c.aliimg.com/fdevlib/css/gallery/animate/special.css"); // Special(其他特殊动画)

addClass

// animated是统一标识，bounceOutLeft是动画类型
$('#element').addClass('animated bounceOutLeft');

duration、delay、times

/* 默认动画持续时间为1s，无delay，执行1次 */
/* vendor为浏览器前缀 */
#element {
    -vendor-animation-duration: 3s;
    -vendor-animation-delay: 2s;
    -vendor-animation-iteration-count: infinite;
}
